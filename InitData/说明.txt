目前数据截至：202003

SELECT CODE_PROV,NAME_PROV FROM [dbo].[ditu] GROUP BY CODE_PROV,NAME_PROV

SELECT CODE_CITY,NAME_CITY,CODE_PROV FROM [dbo].[ditu] GROUP BY CODE_CITY,NAME_CITY,CODE_PROV

SELECT CODE_COUN,NAME_COUN,CODE_CITY FROM [dbo].[ditu] GROUP BY CODE_COUN,NAME_COUN,CODE_CITY

SELECT CODE_TOWN,NAME_TOWN,CODE_COUN FROM [dbo].[ditu] GROUP BY CODE_TOWN,NAME_TOWN,CODE_COUN
